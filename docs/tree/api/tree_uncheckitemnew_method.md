@short: unchecks the checkbox of a tree item and all its sub-items

@signature: uncheckItem(id: string): void;

@params:
- id 	string		the id of an item to uncheck

@example:
// checking an item
tree.checkItem("history");

// unchecking an item 
tree.uncheckItem("history");
