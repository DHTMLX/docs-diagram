@short: sets the minimal width of a sidebar in the collapsed state

@signature: minWidth?: number | string;

@default: "44"

@example: 
var sidebar = new dhx.Sidebar(document.body, {
    css: "dhx_widget--border_right",
    minWidth: 80,
    collapsed: true
});
