@short: sets the number of frozen columns

@signature: {'leftSplit?: number;'}

@example: 
var treegrid = new dhx.TreeGrid("treegrid_container", {
	columns: [// columns config],
	leftSplit:2,  /*!*/
	data: dataset
});
