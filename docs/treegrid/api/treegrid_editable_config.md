@short: enables editing in TreeGrid columns

@signature: {'editable?: boolean;'}

@default: false

@example:
var treegrid = new dhx.TreeGrid("treegrid_container", {
	columns: [// columns config],
	editable:true, /*!*/
	data: dataset
});



@descr:
