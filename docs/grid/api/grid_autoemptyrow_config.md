---          

@short: adds an empty row after the last filled row in the Grid

@signature: {'autoEmptyRow?: boolean;'}

@default: true

@example: 
var grid = new dhx.Grid("grid_container", {
	columns: [// columns config],
	autoEmptyRow:true,  /*!*/
	data: dataset
});
