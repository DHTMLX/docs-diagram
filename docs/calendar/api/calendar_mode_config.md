@short: the mode of Calendar initialization

@signature: {'mode?: ViewMode;'}

@values: "calendar","month","year","timepicker" 

@default: "calendar"

@example:
var calendar = new dhx.Calendar("calendar_container", {
    mode: "month"
});

@descr:
