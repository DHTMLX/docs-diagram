@short: disables a tab on a page

@signature: disableTab(id: string): boolean;

@params:
- id	string  the id of a tab


@returns:
- param	boolean     'true' if the tab is disabled, 'false' otherwise


@example:
tabbar.disableTab("London");
// -> true|false
