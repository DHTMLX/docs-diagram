@short: specifies the mode of displaying a tabbar

@signature: {'mode?: "left" | "right" | "bottom" | "top";'}

@default: "top"

@example:
var tabbar = new dhx.Tabbar("tabbar_container", {
    mode: "bottom"
});

@descr:
