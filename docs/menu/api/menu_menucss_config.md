@short: adds style classes to all containers of Menu controls with nested items 

@signature: menuCss?: string;

@example: 
<style>
    .my_first_class {
        /*some styles*/
    }
 
    .my_second_class {
        /*some styles*/
    }
</style>
 
var menu = new dhx.Menu({
    menuCss:"my_first_class my_second_class"
});


@template:	api_config
@descr:
